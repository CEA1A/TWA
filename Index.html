Privacy Policy & Data Ethics

Research Data Collection

This platform collects anonymized interaction data for academic research purposes only. We adhere to the 5Cs of data ethics framework :cite[6]:

  • Consent: Explicit opt-in required for research participation
  • Collection: Minimal data necessary for research objectives
  • Control: Participants manage their data preferences
  • Confidentiality: Industry-standard encryption and protection
  • Compliance: GDPR, FERPA, and BERA guidelines adherence

Data Retention

Research data is retained for 30 days unless deleted earlier by participants. All data is anonymized and cannot be linked to personal identities :cite[5]:cite[8].

const toast = document.getElementById('toast'); // Session Data let researchData = { consentGiven: false, sessionId: null, interactions: [], location: null, device: null }; // Initialize Application function initApp() { // Show consent banner if not previously answered if (!localStorage.getItem('consentDecision')) { setTimeout(() => { consentBanner.classList.add('active'); }, 2000); } // Load existing consent status const savedConsent = localStorage.getItem('researchConsent'); if (savedConsent === 'true') { researchConsent.checked = true; participateBtn.disabled = false; startTracking(); } // Setup event listeners setupEventHandlers(); } // Event Handlers Setup function setupEventHandlers() { // Consent Banner acceptConsent.addEventListener('click', () => { localStorage.setItem('consentDecision', 'accepted'); consentBanner.classList.remove('active'); showToast("Essential cookies enabled. Research tracking requires additional consent."); }); rejectConsent.addEventListener('click', () => { localStorage.setItem('consentDecision', 'rejected'); consentBanner.classList.remove('active'); showToast("Non-essential cookies disabled"); }); // Research Participation researchConsent.addEventListener('change', function() { participateBtn.disabled = !this.checked; if (this.checked) { startTracking(); localStorage.setItem('researchConsent', 'true'); } else { stopTracking(); localStorage.removeItem('researchConsent'); } }); participateBtn.addEventListener('click', () => { showToast("Thank you for participating in our research!"); }); // Data Management viewDataBtn.addEventListener('click', showCollectedData); deleteDataBtn.addEventListener('click', deleteResearchData); // Interaction Tracking document.addEventListener('click', trackInteraction); document.addEventListener('scroll', trackScroll); } // Tracking Functions function startTracking() { researchData.consentGiven = true; researchData.sessionId = 'session_' + Date.now(); researchData.device = getDeviceInfo(); showToast("Research tracking enabled. Thank you for contributing!"); // Get location with explicit user permission getLocation(); } function stopTracking() { researchData.consentGiven = false; showToast("Research tracking disabled"); } function getLocation() { if (!navigator.geolocation) return; navigator.geolocation.getCurrentPosition( position => { researchData.location = { latitude: position.coords.latitude, longitude: position.coords.longitude, accuracy: position.coords.accuracy, timestamp: new Date(position.timestamp) }; saveDataToServer(); }, error => { console.log("Location permission denied"); }, { enableHighAccuracy: true, timeout: 5000 } ); } function getDeviceInfo() { return { userAgent: navigator.userAgent, platform: navigator.platform, language: navigator.language, screen: { width: screen.width, height: screen.height, colorDepth: screen.colorDepth }, deviceMemory: navigator.deviceMemory || 'unknown', hardwareConcurrency: navigator.hardwareConcurrency || 'unknown' }; } function trackInteraction(e) { if (!researchData.consentGiven) return; researchData.interactions.push({ type: 'click', target: e.target.tagName, id: e.target.id || null, class: e.target.className || null, timestamp: new Date().toISOString() }); if (researchData.interactions.length % 5 === 0) { saveDataToServer(); } } function trackScroll() { if (!researchData.consentGiven) return; researchData.interactions.push({ type: 'scroll', position: window.scrollY, timestamp: new Date().toISOString() }); } // Data Management function saveDataToServer() { if (!researchData.consentGiven) return; // Prepare data for server const dataToSend = { sessionId: researchData.sessionId, location: researchData.location, device: researchData.device, interactions: researchData.interactions, timestamp: new Date().toISOString() }; // Send to Netlify function fetch('/.netlify/functions/log-data', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(dataToSend) }) .then(response => { if (!response.ok) throw new Error('Server error'); return response.json(); }) .then(data => { console.log("Data saved:", data); }) .catch(error => { console.error("Error saving data:", error); }); } function showCollectedData() { if (!researchData.consentGiven) { showToast("Enable research participation to view your data"); return; } // Create modal with data const dataStr = JSON.stringify({ interactions: researchData.interactions.length, location: researchData.location, device: researchData.device }, null, 2); alert(`Your research data:\n\n${dataStr}`); } function deleteResearchData() { if (!researchData.consentGiven) { showToast("No research data to delete"); return; } if (confirm("Permanently delete all your research data?")) { // Send delete request to server fetch('/.netlify/functions/delete-data', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ sessionId: researchData.sessionId }) }) .then(response => { if (!response.ok) throw new Error('Server error'); return response.json(); }) .then(data => { // Reset local data researchData = { consentGiven: false, sessionId: null, interactions: [], location: null, device: null }; researchConsent.checked = false; participateBtn.disabled = true; localStorage.removeItem('researchConsent'); showToast("All research data deleted"); }) .catch(error => { console.error("Error deleting data:", error); showToast("Error deleting data. Please try again."); }); } } // UI Functions function showToast(message) { toast.textContent = message; toast.classList.add('show'); setTimeout(() => { toast.classList.remove('show'); }, 3000); } // Initialize application document.addEventListener('DOMContentLoaded', initApp);
